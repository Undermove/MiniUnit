public class TestDiscoverer {
    public void DiscoverTests(string assemblyPath) {
        var asm = Assembly.LoadFrom(assemblyPath);
        var allTestTypes = asm.GetTypes();
 
        foreach (var testType in allTestTypes) {
            var tests = testType.GetMethods()
                .Where(m => m.GetCustomAttribute<TestAttribute>() != null);
 
            foreach (var m in tests)
                SendTestCase(new TestCase(testType, m));
        }
    }
}

TestExecutor –≤ VSTest

public class TestExecutor {
    public void ExecuteTests(IEnumerable<TestCase> tests) {
        foreach (var test in tests) {
            var instance = Activator.CreateInstance(test.TestType);
            try {
                Invoke(instance, setUp);
                Invoke(instance, test.TestMethod);
                Invoke(instance, tearDown);
                RecordResult(test, TestOutcome.Passed);
            } catch (Exception ex) {
                RecordResult(test, TestOutcome.Failed, ex.Message);
            }
        }
    }
}

            new TestNodeUpdateMessage(req.Session.SessionUid, node));
    }
}
